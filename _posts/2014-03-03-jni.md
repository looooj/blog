new double object

    jclass doubleClass = env->FindClass("java/lang/Double");

    jmethodID doubleCreator = env->GetMethodID( doubleClass, "<init>", "(D)V" );
    jdouble v = 123.123;
    jobject doubleObject = env->NewObject( doubleClass, doubleCreator, v );
